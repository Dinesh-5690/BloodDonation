CREATE TABLE donor_tbl (
    donor_id            VARCHAR2(10) PRIMARY KEY,
    full_name           VARCHAR2(150) NOT NULL,
    gender              VARCHAR2(10),
    age                 NUMBER(3),
    blood_group         VARCHAR2(5),
    last_donation_date  DATE,
    eligible_flag       VARCHAR2(5),
    mobile              VARCHAR2(20),
    email               VARCHAR2(120),
    city                VARCHAR2(80),
    status              VARCHAR2(20)
);
CREATE TABLE donation_appt_tbl (
    appointment_id      NUMBER(10) PRIMARY KEY,
    donor_id           VARCHAR2(10),
    camp_name          VARCHAR2(150),
    camp_location      VARCHAR2(150),
    appointment_date   DATE,
    time_slot          VARCHAR2(20),
    registration_date  DATE,
    units_planned      NUMBER(3),
    units_collected    NUMBER(3),
    appointment_status VARCHAR2(20),
    remarks            VARCHAR2(400),

    CONSTRAINT fk_donor
    FOREIGN KEY (donor_id)
    REFERENCES donor_tbl(donor_id)
);
